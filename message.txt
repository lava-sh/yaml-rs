We also need to fix the tests: [https://github.com/lava-sh/yaml-rs/blob/main/tests/test\_loads.py#L192-L199](https://github.com/lava-sh/yaml-rs/blob/main/tests/test_loads.py#L192-L199)

I think we should consider exactly which formats we can actually parse. For example:

2001-12-14 21:59:43.10  -05
#                     ^^
#                     2

if a date contains two or more spaces, it would be better not to parse it.

YAML 1.2.2 spec on [website](https://yaml.org/spec/1.2.2) mentions the following formats:

-   [Example 2.8 Play by Play Feed from a Game](https://yaml.org/spec/1.2.2/#:~:text=Example%202%2E8%20Play%20by%20Play%20Feed%20from%20a%20Game)
-   [Example 2.22 Timestamps](https://yaml.org/spec/1.2.2/#:~:text=Example%202%2E22%20Timestamps)

canonical: 2001-12-15T02:59:43.1Z
iso8601: 2001-12-14t21:59:43.10-05:00
spaced: 2001-12-14 21:59:43.10 -5
date: 2002-12-14
Date: 2001-11-23 15:03:17 -5
time: 20:03:20

date: \[0000-9999\]-\[1-12\]-\[01-31\]
# 2002-12-14

time: \[00-23\]:\[00-59\]:\[00-59\](.\[0-9\]+)?
# 20:03:20
# 20:00:00.1
# 20:00:00.1000
# 20:20:00.12345
# 20:00:20.999999999999999999

datetime:
\[0000-9999\]-\[1-12\]-\[1-31\]
(
  \[T | t |' '\]
  \[00-23\]:\[00-59\]:\[00-59\]
  (.\[0-9\]+)?
  (
    Z
    | \[+-\]\[00-23\](:\[00-59\])?
    | ' '\[+-\]\[00-23\]
  )?
)?

#

    // SAFETY: bytes.len() >= 10.
    if unsafe { *bytes.get_unchecked(4) != b'-' || *bytes.get_unchecked(7) != b'-' } {
        return Ok(None);

    let day = parse_digits(bytes, 8, 2) as u8;
    let month = parse_digits(bytes, 5, 2) as u8;
    let year = parse_digits(bytes, 0, 4).cast_signed();

    if bytes.len() == 10 {

    }

    if !matches!(sep_byte, b'T' | b't' | b' ' | b'\t') { return Ok(None); }

pub fn parse_py_datetime<'py>(py: Python<'py>, str: &str) -> PyResult<Option<Bound<'py, PyAny>>> {
    let bytes = str.as_bytes();
    if bytes.len() < 10 { return Ok(None); }
    
    // SAFETY: bytes.len() >= 10
    if unsafe { *bytes.get_unchecked(4) != b'-' || *bytes.get_unchecked(7) != b'-' } {
        return Ok(None);
    }
    
    let day = parse_digits(bytes, 8, 2) as u8;
    let month = parse_digits(bytes, 5, 2) as u8;
    let year = parse_digits(bytes, 0, 4).cast_signed();
    
    if bytes.len() == 10 {
        return Ok(Some(PyDate::new(py, year, month, day)?.into_any()));
    }
    
    // Ровно один T/t/SP/TAB на позиции 10
    // SAFETY: bytes.len() > 10
    let sep_byte = unsafe { *bytes.get_unchecked(10) };
    if !matches!(sep_byte, b'T' | b't' | b' ' | b'\t') { return Ok(None); }
    
    let time_start = 11usize;

    if unsafe { *bytes.get_unchecked(time_start + 2) } != b':' { return Ok(None); }
